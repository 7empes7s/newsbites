For football-data.org (10 req/min limit): add request coalescing so multiple server renders of the same page don't each make a separate API call. Use a module-level `Map` to cache in-flight requests:

```typescript
const inFlight = new Map<string, Promise<unknown>>();

export async function fetchWithCoalescing(key: string, fetcher: () => Promise<unknown>) {
  if (inFlight.has(key)) return inFlight.get(key);
  const promise = fetcher().finally(() => inFlight.delete(key));
  inFlight.set(key, promise);
  return promise;
}
```

## Phase 46 — Error Boundaries + Skeleton States

**File: `components/article-panel/PanelErrorBoundary.tsx`**

```typescript
'use client';
import { Component } from 'react';

export class PanelErrorBoundary extends Component<
  { children: React.ReactNode },
  { hasError: boolean }
> {
  constructor(props: { children: React.ReactNode }) {
    super(props);
    this.state = { hasError: false };
  }
  static getDerivedStateFromError() { return { hasError: true }; }
  render() {
    if (this.state.hasError) return null; // silent failure
    return this.props.children;
  }
}
```

## Phase 49 — localStorage Panel Preferences

**File: `lib/panel-prefs.ts`**

```typescript
type PanelPrefs = {
  defaultExpanded: boolean;
  hiddenSections: string[];     // section ids to hide
  pinnedTab: Record<string, string>; // vertical → preferred tab id
};

export function getPanelPrefs(): PanelPrefs {
  try {
    return JSON.parse(localStorage.getItem('newsbites-panel-prefs') || '{}');
  } catch { return {}; }
}
export function setPanelPrefs(prefs: Partial<PanelPrefs>): void {
  localStorage.setItem('newsbites-panel-prefs', JSON.stringify({
    ...getPanelPrefs(), ...prefs,
  }));
}
```

- `PanelDrawer` reads `defaultExpanded` to set its initial open state
- `PanelTabBar` reads `pinnedTab` to restore last-viewed tab per vertical
